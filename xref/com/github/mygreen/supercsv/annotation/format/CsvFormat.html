15   * 任意のタイプの書式を指定するためのアノテーションです。
16   * <p>このライブラリに対応していない書式や独自のタイプに対応する際に使用します。</p>
17   * 
18   * <h3 class="description">基本的な使い方</h3>
19   * <ul>
20   *   <li>属性{@link #formatter()}で、{@link TextFormatter}の実装クラスを指定します。</li>
21   *   <li>{@link TextFormatter}は、文字列とオブジェクトの相互に変換するためのインタフェースになります。
22   *    <br>通常は、抽象クラス{@link AbstractTextFormatter}を継承して作成します。
23   *   </li>
24   * </ul>
25   * 
26   * <p>以下に、{@literal java.net.URL}にマッピングする例を示します。</p>
27   * 
28   * <pre class="highlight"><code class="java">
29   * {@literal @CsvBean}
30   * public class SampleCsv {
31   *     
32   *     {@literal @CsvColumn(number=1, label="ホームページ")}
33   *     {@literal @CsvFormat(formatter=UrlFormatter.class)}  // 実装したTextFormatterの指定
34   *     private URL homepage;
35   *     
36   *     // getter/setterは省略
37   * }
38   * 
39   * // TextFormatterの実装クラス - URLにマッピングするクラス
40   * public class UrlFormatter extends AbstractTextFormatter{@literal <URL>} {
41   *       
42   *       // 読み込み時の文字列からオブジェクト型に変換するメソッドの実装
43   *       {@literal @Override}
44   *       public URL parse(final String text) {
45   *           
46   *           try {
47   *               return new URL(text);
48   *           } catch(MalformedURLException e) {
49   *               throw new TextParseException(text, URL.class, e);
50   *           }
51   *           
52   *       }
53   *       
54   *       // 書き込みにオブジェクト型から文字列に変換するメソッドの実装
55   *       {@literal @Override}
56   *       public String print(final URL object) {
57   *           return object.toExternalForm();
58   *       }
59   *       
60   *       // 読み込み時のエラーメッセージ中で使用可能な変数の定義
61   *       // 必要があればオーバライドして実装します。
62   *       {@literal @Override}
63   *       public {@literal Map<String, Object>} getMessageVariables() {
64   *       
65   *           final {@literal Map<String, Object>} vars = new {@literal HashMap<>}();
66   *           
67   *           vars.put("key", "vars");
68   *           
69   *           return vars;
70   *       }
71   *       
72   *   }
73   * </code></pre>
74   * 
75   * @since 2.0
76   * @author T.TSUCHIE
77   *
78   */

88      @SuppressWarnings("rawtypes")

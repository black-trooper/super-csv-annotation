11  /**
12   * 合成のアノテーションを作成する際に、アノテーションの属性を特定して上書きするためのアノテーションです。
13   * 
14   * <h3 class="description">基本的な使い方</h3>
15   * <ul>
16   *   <li>属性{@link #annotation()}で上書き対象のアノテーションを指定します。</li>
17   *   <li>属性{@link #name()}で上書き対象のアノテーションの属性名を指定します。</li>
18   *   <li>上書きする属性のクラスタイプは合わせる必要があります。</li>
19   *   <li>上書き対象のアノテーションが複数指定されている場合、属性{@link #index()}で指定します。
20   *     <ul>
21   *       <li>値0から始まります。</li>
22   *       <li>インデックスを指定しない場合は、該当するアノテーションの属性が全て上書きされます。</li>
23   *     </ul>
24   *   </li>
25   * </ul>
26   * 
27   * <pre class="highlight"><code class="java">
28   * {@literal @Target({ElementType.FIELD, ElementType.ANNOTATION_TYPE})}
29   * {@literal @Retention(RetentionPolicy.RUNTIME)}
30   * {@literal @Documented}
31   * {@literal @Repeatable(CsvCustomComposition.List.class)}
32   * {@literal @CsvComposition}  // 合成のアノテーションであることを示すためのメタアノテーション
33   * {@literal @CsvDefaultValue(value="0", cases=BuildCase.Read)}
34   * {@literal @CsvRequire}
35   * {@literal @CsvNumberRange(min="0", max="100,000,000", groups=NormalGroup.class)}
36   * {@literal @CsvNumberRange(min="0", max="100,000,000,000", groups=ManagerGroup.class)} // 上書き対象のアノテーション
37   * public {@literal @interface} CsvCustomComposition {
38   *     
39   *     // 2番目（インデックスが1）の{@literal @CsvNumberRange}の属性maxの上書き
40   *     {@literal @CsvOverridesAttribute(annotation=CsvNumberRange.class, name="max", index=1)}
41   *     String managerSalaryMax() default "100,000,000,000,000";
42   *     
43   *     // 繰り返しのアノテーションの格納用アノテーションの定義
44   *     {@literal @Target({ElementType.FIELD, ElementType.ANNOTATION_TYPE})}
45   *     {@literal @Retention(RetentionPolicy.RUNTIME)}
46   *     {@literal @Documented}
47   *     {@literal @interface} List {
48   *     
49   *         CsvCustomComposition[] value();
50   *     }
51   * }
52   * </code></pre>
53   * 
54   * @since 2.0
55   * @author T.TSUCHIE
56   *
57   */
58  @Target({ElementType.METHOD})
59  @Retention(RetentionPolicy.RUNTIME)
60  @Documented
61  @Repeatable(CsvOverridesAttribute.List.class)
62  public @interface CsvOverridesAttribute {
63      
64      /**
65       * 上書き対象のアノテーションのクラスタイプを指定します。
66       * @return 上書き対象のアノテーションのクラスタイプ
67       */
68      Class<? extends Annotation> annotation();
69      
70      /**
71       * 上書き対象のアノテーションの属性名を指定します。
72       * <p>省略した場合、付与した属性名が採用されます。</p>
73       * @return 上書き対象のアノテーションの属性名。
74       */
75      String name() default "";
76      
77      /**
78       * 複数同じアノテーションが指定されている場合に、インデックス番号で特定するために指定します。
79       * @return 指定する場合は0から始めます。
80       */
81      int index() default -1;
82      
83      /**
84       * アノテーションを複数個指定する際の要素です。
85       */
86      @Target({ElementType.METHOD})
87      @Retention(RetentionPolicy.RUNTIME)
88      @Documented
89      @interface List {
90          
91          CsvOverridesAttribute[] value();
92      }

13  /**
14   * 左側にパディングするためのアノテーションです。
15   * 
16   * <h3 class="description">基本的な使い方</h3>
17   * 
18   * <ul>
19   *   <li>属性 {@link #size()}で、パディングするサイズを指定します。
20   *     <br>変換対象の文字長が属性{@link #size()}の値を超える場合、パディングは行いません。
21   *   </li>
22   *   <li>属性{@link #padChar()}でパディング文字を指定することができます。
23   *     <br>デフォルトでは、半角空白がパディング文字です。
24   *   </li>
25   *   <li>アノテーション{@link CsvTrim}と併用する際には、処理結果が互いに変換対象となるため、
26   *      属性{@link #cases()}や{@link #groups()}で適用するケースを分けるようにしてください。
27   *   </li>
28   * </ul>
29   * 
30   * <pre class="highlight"><code class="java">
31   * {@literal @CsvBean}
32   * public class SampleCsv {
33   *     
34   *     // 値が[10]の場合、結果は、[&nbsp;&nbsp;&nbsp;10]となります。
35   *     {@literal @CsvColumn(number=1)}
36   *     {@literal @CsvLefPad(size=5)}
37   *     private Integer id;
38   *     
39   *     // パディング文字を変更する場合
40   *     {@literal @CsvColumn(number=2)}
41   *     {@literal @CsvLefPad(size=5, padChar='_')}
42   *     private Integer number;
43   *     
44   *     // 書き込み時のみ適用する場合
45   *     {@literal @CsvColumn(number=3)}
46   *     {@literal @CsvTrim(cases=BuildCase.Read)}
47   *     {@literal @CsvLefPad(size=20, cases=BuildCase.Write)}
48   *     private String name;
49   *     
50   *     // getter/setterは省略
51   * }
52   * </code></pre>
53   * 
54   *
55   *
56   * @since 2.0
57   * @author T.TSUCHIE
58   *
59   */
60  @Target({ElementType.FIELD, ElementType.ANNOTATION_TYPE})

65  public @interface CsvLeftPad {

71       */
72      int size();
73      
74      /**
75       * パディングする際の文字を指定します。
76       * @return パディングの文字
77       */
78      char padChar() default ' ';
79      
80      /**
81       * 適用するケースを指定します。
82       * @return 何も指定しない場合は全てのケースに適用されます。
83       */
84      BuildCase[] cases() default {};
85      
86      /**
87       * グループのクラスを指定します。
88       * <p>処理ごとに適用するアノテーションを切り替えたい場合に指定します。
89       * @return 指定しない場合は、{@link DefaultGroup}が適用され全ての処理に適用されます。
90       */
91      Class<?>[] groups() default {};
92      
93      /**
94       * アノテーションの処理順序の定義。
95       * @return 値が大きいほど後に実行されます。

97       */
98      int order() default 0;

103     @Target({ElementType.FIELD, ElementType.ANNOTATION_TYPE})

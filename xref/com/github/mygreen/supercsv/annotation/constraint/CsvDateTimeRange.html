16  /**
17   * 値が指定した期間内かどうか検証するためのアノテーションです。
18   * <p>日時型に指定可能です。</p>
19   * 
20   * <h3 class="description">基本的な使い方</h3>
21   * 
22   * <ul>
23   *   <li>属性{@link #min()}で、下限値としての日時を指定します。
24   *     <br>値は、クラスタイプや指定した書式に沿った形式で指定する必要があります。
25   *   </li>
26   *   <li>属性{@link #max()}で、上限値としての日時を指定します。
27   *     <br>値は、クラスタイプや指定した書式に沿った形式で指定する必要があります。
28   *   </li>
29   *   <li>属性{@link #inclusive()}で、値の検証時に属性{@link #min()}と{@link #max()}の値を含むかどうか指定します。
30   *     <br>初期値はtrueで、指定した値を含みます。</li>
31   * </ul>
32   * 
33   * <pre class="highlight"><code class="java">
34   * {@literal @CsvBean}
35   * public class SampleCsv {
36   *     
37   *     // 値は書式に沿った形式で入力します。
38   *     {@literal @CsvColumn(number=1)}
39   *     {@literal @CsvDateTimeFormat(pattern="yyyy/MM/dd")}
40   *     {@literal @CsvDateTimeRange(min="1900/01/01", max="2050/12/31")}
41   *     private LocalDate birthday;
42   *     
43   *     // 比較する際には指定した期間内（境界値を含まない）かどうかで判定します。
44   *     {@literal @CsvColumn(number=2)}
45   *     {@literal @CsvDateTimeRange(min="1970-01-01 00:00:00.000", max="9999-12-31 23:59:59.999", inclusive=false)}
46   *     private Timestamp registerTime;
47   *     
48   *     // getter/setterは省略
49   * }
50   * </code></pre>
51   * 
52   * @since 2.0
53   * @author T.TSUCHIE
54   *
55   */

66       */
67      String min();
68      
69      /**
70       * 上限値としての日時を指定します。
71       * @return 値は、クラスタイプやアノテーションで指定した書式に沿った値を指定する必要があります。
72       */
73      String max();
74      
75      /**
76       * 値を比較する際に指定した値を含むかどうかを指定します。
77       * @return {@code true}の場合、日時以降か、以前かどうかとして比較します。
78       *         {@code false}の場合、値より後か、前かどうかとして比較します。
79       */
80      boolean inclusive() default true;
81      
82      /**
83       * エラー時のメッセージを指定します。
84       * <p>{@literal {key}}の書式の場合、プロパティファイルから取得した値を指定できます。</p>
85       * 
86       * <p>使用可能なメッセージ中の変数は下記の通りです。</p>
87       * <ul>
88       *   <li>lineNumber : カラムの値に改行が含まれている場合を考慮した実際の行番号です。1から始まります。</li>
89       *   <li>rowNumber : CSVの行番号です。1から始まります。</li>
90       *   <li>columnNumber : CSVの列番号です。1から始まります。</li>
91       *   <li>label : カラムの見出し名です。</li>
92       *   <li>validatedValue : 実際のカラムの値です。</li>
93       *   <li>min : アノテーションの属性{@link #min()}をフィールドの型に変換した値です。</li>
94       *   <li>max : アノテーションの属性{@link #max()}をフィールドの型に変換した値です。</li>
95       *   <li>inclusive : アノテーションの属性{@link #inclusive()}の値です。</li>
96       *   <li>printer : カラムの値に対数するフォーマッタです。{@link TextPrinter#print(Object)}でvalidatedValue, min, maxの値を文字列に変換します。</li>
97       * </ul>
98       * 
99       * @return 省略した場合は、適用された{@link CellProcessor}に基づいたメッセージが出力されます。
100      */

16  /**
17   * 値が他の行と異なるか検証するためのアノテーションです。
18   * <p>全てのクラスタイプに指定可能です。</p>
19   * 
20   * <h3 class="description">基本的な使い方</h3>
21   * <p>値を比較する際には、各要素の値のhashCode()メソッドで取得したハッシュ値にて判定します。</p>
22   * 
23   * <pre class="highlight"><code class="java">
24   * {@literal @CsvBean}
25   * public class SampleCsv {
26   *     
27   *     {@literal @CsvColumn(number=1)}
28   *     {@literal @CsvUniqueHashCode}
29   *     private long id;
30   *     
31   *     // getter/setterは省略
32   * }
33   * </code></pre>
34   * 
35   * @since 2.0
36   * @author T.TSUCHIE
37   *
38   */
39  @Target({ElementType.FIELD, ElementType.ANNOTATION_TYPE})
40  @Retention(RetentionPolicy.RUNTIME)
41  @Documented
42  @Repeatable(CsvUniqueHashCode.List.class)
43  @CsvConstraint(value={})
44  public @interface CsvUniqueHashCode {
45      
46      /**
47       * エラー時のメッセージを指定します。
48       * <p>{@literal {key}}の書式の場合、プロパティファイルから取得した値を指定できます。</p>
49       * 
50       * <p>使用可能なメッセージ中の変数は下記の通りです。</p>
51       * <ul>
52       *   <li>lineNumber : カラムの値に改行が含まれている場合を考慮した実際の行番号です。1から始まります。</li>
53       *   <li>rowNumber : CSVの行番号です。1から始まります。</li>
54       *   <li>columnNumber : CSVの列番号です。1から始まります。</li>
55       *   <li>label : カラムの見出し名です。</li>
56       *   <li>validatedValue : 実際のカラムの値です。</li>
57       *   <li>hashCode : valueに対するハッシュ値です。</li>
58       *   <li>printer : カラムの値に対数するフォーマッタです。{@link TextPrinter#print(Object)}でvalidatedValueの値を文字列に変換します。</li>
59       * </ul>
60       * 
61       * @return 省略した場合は、適用された{@link CellProcessor}に基づいたメッセージが出力されます。
62       */
63      String message() default "{com.github.mygreen.supercsv.annotation.constraint.CsvUniqueHashCode.message}";
64      
65      /**
66       * 適用するケースを指定します。
67       * @return 何も指定しない場合は全てのケースに適用されます。
68       */
69      BuildCase[] cases() default {};
70      
71      /**
72       * グループのクラスを指定します。
73       * <p>処理ごとに適用するアノテーションを切り替えたい場合に指定します。
74       * @return 指定しない場合は、{@link DefaultGroup}が適用され全ての処理に適用されます。
75       */
76      Class<?>[] groups() default {};
77      
78      /**
79       * アノテーションの処理順序の定義。
80       * @return 値が大きいほど後に実行されます。
81       *         値が同じ場合は、アノテーションのクラス名の昇順になります。
82       */
83      int order() default 0;

54      private String padding(final String str) {
55          
56          final int pads = padSize - str.length();
57          if(pads <= 0) {

60          
61          final StringBuilder sb = new StringBuilder(str.length() + pads);

63          sb.append(str);
64          
65          for(int i=0; i < pads; i++) {

67          }

71      }

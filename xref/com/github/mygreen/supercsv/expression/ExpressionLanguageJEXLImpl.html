30          this(new JexlEngine());

34       * {@link JexlEngine}を指定するコンストラクタ。
35       * @param jexlEngine JEXLの処理エンジン。

37      public ExpressionLanguageJEXLImpl(final JexlEngine jexlEngine) {
38          this.jexlEngine = jexlEngine;
39      }
40      
41      @Override
42      public Object evaluate(final String expression, final Map<String, Object> values) {
43          
44          Objects.requireNonNull(expression, "expression shoud not be null.");
45          Objects.requireNonNull(values, "values shoud not be null.");
46          
47          if(logger.isDebugEnabled()) {
48              logger.debug("Evaluating JEXL expression: {}", expression);
49          }
50          
51          try {
52              Expression expr = expressionCache.get(expression);
53              if (expr == null) {
54                  expr = jexlEngine.createExpression(expression);
55                  expressionCache.put(expression, expr);
56              }
57              
58              return expr.evaluate(new MapContext((Map<String, Object>) values));
59              
60          } catch(Exception ex) {
61              throw new ExpressionEvaluationException(String.format("Evaluating [%s] script with JEXL failed.", expression), ex,
62                      expression, values);
63          }
64      }

72      protected <T extends Enum<?>> Method getEnumValueMethod(final Class<T> enumClass, final String valueMethodName) {
73          try {
74              final Method method =  enumClass.getMethod(valueMethodName);
75              method.setAccessible(true);
76              return method;
77              
78          } catch (ReflectiveOperationException e) {
79              throw new IllegalArgumentException(String.format("not found method '%s'", valueMethodName), e);
80          }
81          
82      }

84      protected <T extends Enum<T>> Map<Enum<?>, String> createEnumMap(final Class<T> enumClass) {
85          
86          EnumSet<T> set = EnumSet.allOf(enumClass);
87          
88          final Map<Enum<?>, String> map = new LinkedHashMap<>();
89          for(T e : set) {
90              map.put(e, e.name());            
91              
92          }
93          
94          return Collections.unmodifiableMap(map);
95      }
96      
97      protected <T extends Enum<T>> Map<Enum<?>, String> createEnumMap(final Class<T> enumClass, final String methodName) {
98          
99          final Method method = getEnumValueMethod(enumClass, methodName);
100         
101         final Map<Enum<?>, String> map = new LinkedHashMap<>();
102         try {
103             final EnumSet<T> set = EnumSet.allOf(enumClass);
104             for(T e : set) {
105                 Object returnValue = method.invoke(e);
106                 map.put(e, returnValue.toString());            
107                 
108             }
109             
110         } catch(ReflectiveOperationException e) {
111             throw new RuntimeException("fail get enum value.", e);
112         }
113         
114         return Collections.unmodifiableMap(map);
115     }

144     public Map<String, ?> getMessageVariable() {
145         final Map<String, Object> vars = new HashMap<String, Object>();
146         vars.put("type", getType().getCanonicalName());
147         vars.put("valueMethod", getValueMethod() == null ? "" : getValueMethod().getName());
148         
149         final List<Enum<?>> enumValues = getEnumValueMap().entrySet().stream()
150                 .map(e -> e.getKey())
151                 .collect(Collectors.toList());
152         
153         final String enumsStr = getEnumValueMap().entrySet().stream()
154                 .map(e -> e.getValue())
155                 .collect(Collectors.joining(", "));
156         
157         vars.put("enumValues", enumValues);
158         vars.put("enumsStr", enumsStr);
159         
160         return vars;
161     }

109     @SuppressWarnings("unchecked")
110     @Override
111     public Object execute(final Object value, final CsvContext context) {
112         validateInputNotNull(value, context);
113         
114         if(!(value instanceof String)) {
115             throw new SuperCsvCellProcessorException(String.class, value, context, this);
116         }
117         
118         try {
119             Object result = formatter.parse(type, (String) value);
120             return next.execute(result, context);
121             
122         } catch(ParseException | ConversionException e) {
123             throw new SuperCsvCellProcessorException(
124                     String.format("'%s' could not be parsed as a BigDecimal", value),
125                     context, this, e);
126         }
127     }

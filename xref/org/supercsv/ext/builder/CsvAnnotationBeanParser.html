51      public <T> CsvBeanMapping<T> parse(final Class<T> clazz, final boolean ignoreValidationProcessorOnOutput) {
52          
53          if(clazz == null) {
54              throw new IllegalArgumentException("clazz must be not null.");
55          }
56          
57          CsvBeanMapping<T> mappingBean = new CsvBeanMapping<T>(clazz);
58          
59          // @CsvBean
60          CsvBean csvBeanAnno = clazz.getAnnotation(CsvBean.class);
61          if(csvBeanAnno == null) {
62              throw new SuperCsvInvalidAnnotationException("not found annotation 'CsvBean'");
63          }
64          
65          mappingBean.setHeader(csvBeanAnno.header());
66          
67          List<CsvColumnMapping> mappingColumns = new ArrayList<CsvColumnMapping>();
68          
69          // @CsvColumn for all(public / private / default / protected)
70          for(Field field : clazz.getDeclaredFields()) {
71              
72              CsvColumn csvColumnAnno = field.getAnnotation(CsvColumn.class);
73              if(csvColumnAnno == null) {
74                  continue;
75              }
76              
77              mappingColumns.add(createColumnMapping(field, csvColumnAnno, ignoreValidationProcessorOnOutput));
78              
79          }
80          
81          // sorting column data by position value
82          Collections.sort(mappingColumns, new Comparator<CsvColumnMapping>() {
83              
84              @Override
85              public int compare(final CsvColumnMapping o1, final CsvColumnMapping o2) {
86                  if(o1.getPosition() > o2.getPosition()) {
87                      return 1;
88                  } else if(o1.getPosition() == o2.getPosition()) {
89                      return o1.getColumnName().compareTo(o2.getColumnName());
90                  } else {
91                      return -1;
92                  }
93              }
94              
95          });
96          
97          validatePosition(mappingColumns);
98          mappingBean.setColumns(mappingColumns);
99          
100         return mappingBean;
101     }

103     private void validatePosition(final List<CsvColumnMapping> columns) {
104         if(columns.isEmpty()) {
105             throw new SuperCsvInvalidAnnotationException("not found column definition.");
106         }
107         
108         // check duplicated position value 
109         Set<Integer> checkedPosiiton = new TreeSet<Integer>();
110         Set<Integer> duplicatePosition = new TreeSet<Integer>();
111         for(CsvColumnMapping columnMapping : columns) {
112             
113             if(checkedPosiiton.contains(columnMapping.getPosition())) {
114                 duplicatePosition.add(columnMapping.getPosition());
115             }
116             checkedPosiiton.add(columnMapping.getPosition());
117             
118         }
119         
120         // check lacked or skipped position value 
121         Set<Integer> lackPosition = new TreeSet<Integer>();
122         final int maxPosition = columns.get(columns.size()-1).getPosition();
123         for(int i=0; i < maxPosition; i++) {
124             if(!checkedPosiiton.contains(i)) {
125                 lackPosition.add(i);
126             }
127         }
128         
129         if(!lackPosition.isEmpty() || !duplicatePosition.isEmpty()) {
130             throw new SuperCsvInvalidAnnotationException(
131                     String.format("position value is wrong. lacked position=%s, duplicated position=%s",
132                             lackPosition, duplicatePosition));
133         }
134     }

136     @SuppressWarnings({"rawtypes", "unchecked"})
137     private CsvColumnMapping createColumnMapping(final Field field, final CsvColumn csvColumnAnno,
138             final boolean ignoreValidationProcessorOnOutput) {
139         
140         CsvColumnMapping columnMapping = new CsvColumnMapping();
141         
142         columnMapping.setColumnName(field.getName());
143         columnMapping.setColumnType(field.getType());
144         columnMapping.setPosition(csvColumnAnno.position());
145         
146         if(csvColumnAnno.label() != null && !csvColumnAnno.label().isEmpty()) {
147             columnMapping.setLabel(csvColumnAnno.label());
148         } else {
149             columnMapping.setLabel(field.getName());
150         }
151         
152         CellProcessorBuilder builder = null;
153         if(csvColumnAnno.builderClass().equals(DefaultCellProcessorBuilder.class)) {
154             builder = builderContainer.getBuilder(field.getType());
155             
156             // if enum type
157             if(builder == null && Enum.class.isAssignableFrom(field.getType())) {
158                 builder = builderContainer.getBuilder(Enum.class);
159             }
160             
161             // if not found builder, use default builder.
162             if(builder == null) {
163                 builder = DefaultCellProcessorBuilder.INSTANCE;
164             }
165             
166         } else {
167             // use custom builder class
168             try {
169                 builder = builderFactory.create(csvColumnAnno.builderClass());
170             } catch (Throwable e) {
171                 throw new SuperCsvInvalidAnnotationException(
172                         String.format("fail create instance of %s with 'builderClass' of @CsvColumn property",
173                                 csvColumnAnno.builderClass().getCanonicalName()), e);
174             }
175         }
176         
177         if(builder != null) {
178             columnMapping.setOutputCellProcessor(builder.buildOutputCellProcessor(
179                     field.getType(), field.getAnnotations(), ignoreValidationProcessorOnOutput));
180             
181             columnMapping.setInputCellProcessor(builder.buildInputCellProcessor(
182                     field.getType(), field.getAnnotations()));
183             
184         } else {
185             // user default builder.
186             throw new SuperCsvInvalidAnnotationException(
187                     String.format("not resolve CellProecssorBuilder for field '%s#%s' and type '%s'",
188                             field.getDeclaringClass().getName(), field.getName(),
189                             field.getType().getClass().getName()));
190         }
191         
192         return columnMapping;
193         
194     }

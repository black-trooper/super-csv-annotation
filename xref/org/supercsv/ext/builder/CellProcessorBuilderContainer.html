24  public class CellProcessorBuilderContainer {
25      
26      @SuppressWarnings("rawtypes")
27      private final Map<Class<?>, AbstractCellProcessorBuilder> builderMap;
28      
29      @SuppressWarnings("rawtypes")
30      public CellProcessorBuilderContainer() {
31          builderMap = new ConcurrentHashMap<Class<?>, AbstractCellProcessorBuilder>();
32          init();
33      }
34      
35      public void init() {
36          builderMap.clear();
37          
38          registBuilder(String.class, new StringCellProcessorBuilder());
39          
40          registBuilder(Character.class, new CharacterCellProcessorBuilder());
41          registBuilder(char.class, new CharacterCellProcessorBuilder());
42          
43          registBuilder(Boolean.class, new BooleanCellProcessorBuilder());
44          registBuilder(boolean.class, new BooleanCellProcessorBuilder());
45          
46          registBuilder(Byte.class, NumberCellProcessorBuilder.newByte());
47          registBuilder(byte.class, NumberCellProcessorBuilder.newByte());
48          registBuilder(Short.class, NumberCellProcessorBuilder.newShort());
49          registBuilder(short.class, NumberCellProcessorBuilder.newShort());
50          registBuilder(Integer.class, NumberCellProcessorBuilder.newInteger());
51          registBuilder(int.class, NumberCellProcessorBuilder.newInteger());
52          registBuilder(Long.class, NumberCellProcessorBuilder.newLong());
53          registBuilder(long.class, NumberCellProcessorBuilder.newLong());
54          registBuilder(Float.class, NumberCellProcessorBuilder.newFloat());
55          registBuilder(float.class, NumberCellProcessorBuilder.newFloat());
56          registBuilder(Double.class, NumberCellProcessorBuilder.newDouble());
57          registBuilder(double.class, NumberCellProcessorBuilder.newDouble());
58          
59          registBuilder(BigDecimal.class, NumberCellProcessorBuilder.newBigDecimal());
60          registBuilder(BigInteger.class, NumberCellProcessorBuilder.newBigInteger());
61          
62          registBuilder(Date.class, new DateCellProcessorBuilder());
63          registBuilder(java.sql.Date.class, DateCellProcessorBuilder.newSqlDate());
64          registBuilder(Timestamp.class, DateCellProcessorBuilder.newTimestamp());
65          registBuilder(Time.class, DateCellProcessorBuilder.newTime());
66          
67          registBuilder(Enum.class, new EnumCellProcessorBuilder());
68      }
69      
70      public <T> void registBuilder(final Class<?> type, final AbstractCellProcessorBuilder<T> builder) {
71          builderMap.put(type, builder);
72          
73      }
74      
75      @SuppressWarnings("unchecked")
76      public <T> AbstractCellProcessorBuilder<T> getBuilder(final Class<?> type) {
77          return builderMap.get(type);

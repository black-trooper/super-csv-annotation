18  public class ResourceBundleMessageResolver implements MessageResolver {
19      
20      public static final String USER_MESSAGE = "SuperCsvMessages";
21      
22      public static final String DEFAULT_MESSAGE = String.format("%s.%s", ResourceBundleMessageResolver.class.getPackage().getName(), USER_MESSAGE);
23      
24      private final Map<ResourceBundle, List<String>> messageBundleKeys = new HashMap<ResourceBundle, List<String>>(8);
25      
26      private final LinkedList<ResourceBundle> messageBundles = new LinkedList<ResourceBundle>();
27      
28      public static final ResourceBundleMessageResolver INSTANCE = new ResourceBundleMessageResolver();
29      
30      public ResourceBundleMessageResolver() {
31          // add the message bundle for the pre-built constraints in the default locale
32          addMessageBundle(ResourceBundle.getBundle(DEFAULT_MESSAGE));
33          
34          try {
35              // load locale messages.
36              addMessageBundle(ResourceBundle.getBundle(USER_MESSAGE));
37          } catch(Throwable e) { }
38      }
39      
40      /**
41       * {@inheritDoc}
42       */
43      public String getMessage(final String code) {
44          for (final ResourceBundle bundle : messageBundles) {
45              final List<String> keys = messageBundleKeys.get(bundle);
46              if (keys.contains(code)) return bundle.getString(code);
47          }
48          return null;
49      }
50      
51      /**
52       * Adds a message bundle
53       * 
54       * @param messageBundle
55       * @return true if the bundle was registered and false if it was already registered
56       */
57      public final boolean addMessageBundle(final ResourceBundle messageBundle) {
58          if (messageBundles.contains(messageBundle)) return false;
59          
60          messageBundles.addFirst(messageBundle);
61          final List<String> keys = new ArrayList<String>();
62          
63          for (final Enumeration<String> keysEnum = messageBundle.getKeys(); keysEnum.hasMoreElements();) {
64              keys.add(keysEnum.nextElement());
65          }
66          
67          messageBundleKeys.put(messageBundle, keys);
68          
69          return true;
70      }
71      
72      /**
73       * Removes the message bundle
74       * 
75       * @param messageBundle
76       * @return true if the bundle was registered and false if it wasn't registered
77       */
78      public boolean removeMessageBundle(final ResourceBundle messageBundle) {
79          if (!messageBundles.contains(messageBundle)) return false;
80          
81          messageBundles.remove(messageBundle);
82          return true;
83      }
